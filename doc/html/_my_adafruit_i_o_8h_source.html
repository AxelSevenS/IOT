145void timecallback(uint32_t current) {
146
147 // adjust to local time zone
148 current += (timeZone * 60 * 60);
149
150 // calculate current time
151 sec = current % 60;
152 current /= 60;
153 mini = current % 60;
154 current /= 60;
155 hour = current % 24;
156
157 // print hour
158 if(hour == 0 || hour == 12)
159 Serial.print("12");
160 if(hour < 12)
161 Serial.print(hour);
162 else
163 Serial.print(hour - 12);
164
165 // print mins
166 Serial.print(":");
167 if(mini < 10) Serial.print("0");
168 Serial.print(mini);
169
170 // print seconds
171 Serial.print(":");
172 if(sec < 10) Serial.print("0");
173 Serial.print(sec);
174
175 if(hour < 12)
176 Serial.println(" AM");
177 else
178 Serial.println(" PM");
179
180}
