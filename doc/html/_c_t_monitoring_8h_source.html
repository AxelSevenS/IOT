93void device_interact(const char* address, const char* name, long last_contact) {
94
95 MYDEBUG_PRINT("DEVICE : Interacted with ");
96 MYDEBUG_PRINTLN(name);
97
98 read_write_file(MONITORING_FILE_PATH, [=](JsonDocument& monitoringDocument, File& monitoringFile) {
99 auto otherDevice = monitoringDocument[address];
100
101 // mettre à jour le nom du device à chaque interaction, car il peut changer
102 if ( !otherDevice["name"].isNull() != 0 ) {
103 otherDevice["name"] = name;
104 }
105
106 // si c'est la première fois qu'on rencontre le device, assigner le temps de première rencontre
107 auto first_contact = otherDevice["first_contact"];
108 if ( first_contact.isNull() ) {
109 first_contact = getNTP();
110 }
111
112 // mettre à jour le temps d'exposition
113 auto total_exposure = otherDevice["total_exposure"];
114 float exposure_time = total_exposure.as<float>();
115 float time_since_last_contact = (millis() - last_contact) / 1000.0;
116
117 if ( total_exposure.isNull() ) {
118 total_exposure = time_since_last_contact;
119 } else {
120 total_exposure = exposure_time + time_since_last_contact;
121 }
122 exposure_time = total_exposure.as<float>();
123
124 // marquer le device pour surveillance rapprochée si le temps d'exposition est supérieur à la durée de surveillance
125 if (exposure_time > (float)monitoring_time * 60) {
126 otherDevice["close_monitoring"] = true;
127 }
128
129 become_contact_if_device_is_sick(monitoringDocument, monitoringFile, address);
130
131 });
132
133}
