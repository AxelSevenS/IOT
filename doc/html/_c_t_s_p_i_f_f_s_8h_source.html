72void read_file(const char* fileName, FileReadFunction& readFunction) {
73 if (!SPIFFS.begin(true)) {
74 MYDEBUG_PRINTLN("-SPIFFS : Impossible de monter le système de fichiers");
75 return;
76 }
77
78 File jsonFile = SPIFFS.open(fileName, FILE_READ);
79
80 if ( !jsonFile ) {
81 MYDEBUG_PRINTLN("-SPIFFS : Impossible d'ouvrir le fichier en mode lecture");
82 jsonFile.close();
83 SPIFFS.end();
84 return;
85 }
86
87 DynamicJsonDocument jsonDocument(512);
88 DeserializationError error = deserializeJson(jsonDocument, jsonFile);
89
90 if (error){
91 MYDEBUG_PRINT("-SPIFFS : [ERREUR : ");
92 MYDEBUG_PRINT(error.c_str());
93 MYDEBUG_PRINT("] Impossible de parser le JSON dans le fichier : ");
94 MYDEBUG_PRINTLN(jsonFile.name());
95 }
96
97 readFunction(jsonDocument, jsonFile);
98
99 jsonFile.close();
100 SPIFFS.end();
101}

146void update_device_config( const char* new_device_name, const char* new_device_password, JsonDocument& jsonDocument ) {
147
148 bool deviceCredentialsChanged = false;
149
150 if ( strcmp(new_device_name, device_name.c_str()) != 0 ) {
151 deviceCredentialsChanged = true;
152 }
153 device_name = String(new_device_name);
154 jsonDocument["device"]["name"] = device_name;
155
156 if ( strcmp(new_device_password, device_password.c_str()) != 0 ) {
157 deviceCredentialsChanged = true;
158 }
159 device_password = String(new_device_password);
160 jsonDocument["device"]["password"] = device_password;
161
162 if ( deviceCredentialsChanged ) {
163 MYDEBUG_PRINT("Nouveau Nom d'appareil : ");
164 MYDEBUG_PRINT(device_name);
165 MYDEBUG_PRINT(" | Nouveau Mot de Passe : ");
166 MYDEBUG_PRINTLN(device_password);

411 read_write_file(CONFIG_FILE_PATH, [](JsonDocument& jsonDocument, File& jsonFile) {
412
413
414 // On récupère les données du fichier de configuration et on les stocke dans les variables globales

417
418 auto json_name = jsonDocument["device"]["name"];
419 if ( json_name.isNull() ) {
420 MYDEBUG_PRINTLN("-SPIFFS : Pas de nom de device dans le fichier de configuration");
421 json_name = "ContactTracer " + String(random(0xffff), HEX);
422 }
423
424 auto json_password = jsonDocument["device"]["password"];
425 if ( json_password.isNull() ) {
426 MYDEBUG_PRINTLN("-SPIFFS : Pas de mot de passe dans le fichier de configuration");
427 json_password = DEFAULT_PASSWORD;
428 }
429
430 update_device_config(json_name.as<const char*>(), json_password.as<const char*>(), jsonDocument);
431
432
433
434 auto json_status = jsonDocument["status"];
435 if ( json_status.isNull() || json_status == "" ) {
436 MYDEBUG_PRINTLN("-SPIFFS : Pas de statut dans le fichier de configuration");
437 json_status = DEFAULT_USER_STATUS;
438 }
439
440 HealthStatus new_health_status = json_status == "HEALTHY" ? HEALTHY : json_status == "CONTACT" ? CONTACT : SICK;
441
442 update_health_status(new_health_status, jsonDocument);
443
444
445
446 auto json_net_ssid = jsonDocument["wifi"]["net_ssid"];

448 MYDEBUG_PRINTLN("-SPIFFS : Pas de SSID NET dans le fichier de configuration");

455 json_net_password = DEFAULT_NET_PASSWORD;

459
460
461
462 auto json_monitoring_time = jsonDocument["monitoring"]["time"];
463 if ( json_monitoring_time.isNull() ) {
464 MYDEBUG_PRINTLN("-SPIFFS : Pas d'intervalle de surveillance dans le fichier de configuration");
465 json_monitoring_time = DEFAULT_MONITORING_TIME;
