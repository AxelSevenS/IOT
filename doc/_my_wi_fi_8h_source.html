103void wifiScan(){
104 // WiFi.scanNetworks scanne les SSID accessibles, les stocke en mémoire et retourne le nombre
105 // de réseaux trouvés pour qu'on puisse les parcourir.
106 int n = WiFi.scanNetworks();
107 MYDEBUG_PRINT("Scan done");
108 if (n == 0) {
109 // Aucun SSID n'a été trouvé
110 Serial.println("no networks found");
111 } else {
112 // Boucle d'affichage de tous les SSID trouvés et leurs informations
113 Serial.print(n);
114 Serial.println(" networks found");
115 Serial.println("Nr | SSID | RSSI | CH | Encryption");
116 for (int i = 0; i < n; ++i) {
117 // Print SSID and RSSI for each network found
118 Serial.printf("%2d",i + 1);
119 Serial.print(" | ");
120 Serial.printf("%-32.32s", WiFi.SSID(i).c_str());
121 Serial.print(" | ");
122 Serial.printf("%4d", WiFi.RSSI(i));
123 Serial.print(" | ");
124 Serial.printf("%2d", WiFi.channel(i));
125 Serial.print(" | ");
126 switch (WiFi.encryptionType(i))
127 {
128 case WIFI_AUTH_OPEN:
129 Serial.print("open");
130 break;
131 case WIFI_AUTH_WEP:
132 Serial.print("WEP");
133 break;
134 case WIFI_AUTH_WPA_PSK:
135 Serial.print("WPA");
136 break;
137 case WIFI_AUTH_WPA2_PSK:
138 Serial.print("WPA2");
139 break;
140 case WIFI_AUTH_WPA_WPA2_PSK:
141 Serial.print("WPA+WPA2");
142 break;
143 case WIFI_AUTH_WPA2_ENTERPRISE:
144 Serial.print("WPA2-EAP");
145 break;
146 case WIFI_AUTH_WPA3_PSK:
147 Serial.print("WPA3");
148 break;
149 case WIFI_AUTH_WPA2_WPA3_PSK:
150 Serial.print("WPA2+WPA3");
151 break;
152 case WIFI_AUTH_WAPI_PSK:
153 Serial.print("WAPI");
154 break;
155 default:
156 Serial.print("unknown");
157 }
158 Serial.println();
159 delay(10);
160 }
161 }
162 Serial.println("");
163
164 // Effacement des résultats du scan de la mémoire, pour optimiser.
165 WiFi.scanDelete();
166}
